# MAC安装Oracle

# 一. 安装Docker

https://download.docker.com/mac/stable/Docker.dmg

# 1.设置国内镜像源

在 Preferences -> Daemon -> Registry mirrors 写入

https://hh84nuy2.mirror.aliyuncs.com

完成之后， 点击 Apply & Restart

或者添加一个文件 /etc/docker/daemon.json : 写入内容：

{
  "registry-mirrors": ["https://hh84nuy2.mirror.aliyuncs.com"]
}

完成之后 重启docker服务

# 2. 获取 Oracle镜像

docker pull alexeiled/docker-oracle-xe-11g

# 3.启动Oracle

docker run  -d -p 10022:22 -p 11521:1521 -p 18080:8080 alexeiled/docker-oracle-xe-11g
  • --name: 指定名称
  • -h: hostname 主机名
  • -d 后台运行
  • -p ： 宿主机 -->docker 宿主机端口映射

# 4.连接Oracle

oracle端口： 11521 , 用户名 system 、密码 oracle, 连接的URL

jdbc:oracle:thin:@localhost:11521:XE

注意监听是 XE

# 三. 保存当前镜像

因为当前镜像已经做了修改（创建用户，导入过数据）， 所以需要保存更改。

docker  commit  b665da799a4a   jats003-oracle

保存之后， 下次启动只需要启动 jats003/oracle 即可

docker run -d -p 10022:22 -p 11521:1521 -p 18080:8080 jats003-oracle

# 四. 无wm_concat问题

解决方法如下

# 1. 方案1：官方脚本

  1. 创建 wmsys 用户（如果该用户不存在）

    -- 创建用户 
CREATE USER wmsys IDENTIFIED BY wmsys;

--授权用户
GRANT CONNECT,RESOURCE TO wmsys;

  2. 执行官方补丁脚本 使用管理员登录

    sqlplus  system/oracle

    执行脚本：【下载地址：百度云 (opens new window) ， 密码：6688 】 先将脚本文件拷贝到docker容器中：

   docker  cp  本机脚本文件路径     容器ID:/root

容器ID 使用 docker ps 命令查看

   SQL> @/root/owmctab.plb
   SQL> @/root/owmaggrs.plb
   SQL> @/root/owmaggrb.plb

# 2. 方案2：自定义脚本

如果官方脚本执行失败， 则自己手动创建

  1. 创建 wmsys 用户（如果该用户不存在）

    -- 创建用户 
CREATE USER wmsys IDENTIFIED BY wmsys ; 

--授权用户
GRANT CONNECT,RESOURCE TO wmsys;

  2. 创建wm_concat函数

    -- 创建类型
CREATE OR REPLACE TYPE wm_concat_impl   
  AUTHID CURRENT_USER
AS OBJECT (
   curr_str   VARCHAR2 (32767),
   STATIC FUNCTION odciaggregateinitialize (sctx IN OUT wm_concat_impl)
      RETURN NUMBER,
   MEMBER FUNCTION odciaggregateiterate (
      SELF   IN OUT   wm_concat_impl,
      p1     IN       VARCHAR2
   )
      RETURN NUMBER,
   MEMBER FUNCTION odciaggregateterminate (
      SELF          IN       wm_concat_impl,
      returnvalue   OUT      VARCHAR2,
      flags         IN       NUMBER
   )
      RETURN NUMBER,
   MEMBER FUNCTION odciaggregatemerge (
      SELF    IN OUT   wm_concat_impl,
      sctx2   IN       wm_concat_impl
   )
      RETURN NUMBER
);
/

CREATE OR REPLACE TYPE BODY wm_concat_impl
IS
   STATIC FUNCTION odciaggregateinitialize (sctx IN OUT wm_concat_impl)
      RETURN NUMBER
   IS
   BEGIN
      sctx := wm_concat_impl (NULL);
      RETURN odciconst.success;
   END;
   MEMBER FUNCTION odciaggregateiterate (
      SELF   IN OUT   wm_concat_impl,
      p1     IN       VARCHAR2
   )
      RETURN NUMBER
   IS
   BEGIN
      IF (curr_str IS NOT NULL)
      THEN
         curr_str := curr_str || ',' || p1;
      ELSE
         curr_str := p1;
      END IF;

      RETURN odciconst.success;
   END;
   MEMBER FUNCTION odciaggregateterminate (
      SELF          IN       wm_concat_impl,
      returnvalue   OUT      VARCHAR2,
      flags         IN       NUMBER
   )
      RETURN NUMBER
   IS
   BEGIN
      returnvalue := curr_str;
      RETURN odciconst.success;
   END;
   MEMBER FUNCTION odciaggregatemerge (
      SELF    IN OUT   wm_concat_impl,
      sctx2   IN       wm_concat_impl
   )
      RETURN NUMBER
   IS
   BEGIN
      IF (sctx2.curr_str IS NOT NULL)
      THEN
         SELF.curr_str := SELF.curr_str || ',' || sctx2.curr_str;
      END IF;

      RETURN odciconst.success;
   END;
END;
/

CREATE OR REPLACE FUNCTION wm_concat (p1 VARCHAR2)
   RETURN VARCHAR2
   AGGREGATE USING wm_concat_impl;
/

    将wm_concat 授权给所有人使用

    grant execute on wm_concat to public;

# 关于数据持久化

问题来源：

每次启动容器， 对数据库做修改， 比如新增表， 再 重启容器 之后会导致 数据丢失 。

解决办法， 启动容器， 将 容器Oracle安装目录 所有数据 拷贝到本地

docker  cp  容器ID:/u01/app/oracle    /Users/willon/Software/oracle_data/oracle

拷贝之后， 停止容器， 重新运行：

docker  run  -p 11521:1521  -p 10022:22 -p 18080:8080    --name   oracle   -v  /Users/willon/Software/oracle_data/oracle:/u01/app/oracle  jats003/oracle

-v 指定挂载卷 本地卷 ：容器卷

下次重启之后， 可以根据容器名称运行， 且可以实现持久化

docker start oracle
